        // 点数配列を宣言 ＆ 初期データを投入
        let scores = [85, 95, 85, 75]

        // 合計点を、@{for...of} 文 を使って求める
        let total = 0
        for (const score of scores) {
          total += score
        }

        // 平均点を求める
        let average = total / scores.length

        // 結果を表示する
        console.log(total)
        console.log(average)
